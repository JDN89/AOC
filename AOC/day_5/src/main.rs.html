1 use nom::combinator::all_consuming;
 2 use nom::Finish;
 3 
 4 use crate::parse::{Instruction, parse_instruction, parse_row};
 5 use crate::transpose::transpose_rev;
 6 
 7 mod parse;
 8 mod transpose;
 9 
10 fn main() {
11     let mut lines = include_str!("input.txt").lines();
12 
13     let crate_lines: Vec<_> = lines
14         .by_ref()
15         .map_while(|line| {
16             all_consuming(parse_row)(line)
17                 .finish()
18                 .ok()
19                 .map(|(_, line)| line)
20         })
21         .collect();
22     let mut crate_columns = transpose_rev(crate_lines);
23     for col in &crate_columns {
24         println!("{:?}", col);
25     }
26 
27     assert!(lines.next().unwrap().is_empty());
28 
29     let instructions: Vec<Instruction> = lines
30         .map(|line|
31             all_consuming(parse_instruction)(line).finish().unwrap().1)
32         .collect();
33     for ins in &instructions {
34             for _p in 0..ins.mve {
35                let temp =  crate_columns[ins.from].pop();
36                 println!("temp: {:?} ",temp);
37                 match temp {
38                     Some(t) => crate_columns[ins.to].push(t),
39                     None => println!("stack is empty")
40                 }
41 
42             }
43         }
44     println!("result: {:?}", crate_columns);
45     let mut result:Vec<&char> = vec![];
46     let _ = crate_columns.iter().for_each(
47         |vec| match vec.last() {
48             Some(last_value) => result.push(last_value),
49             //println!("last: {:?}",last_value) ,
50             None => println!("no value"),
51         }
52     );
53     println!("result = {:?}",result);
54 
55     }
